<script>
const STORE = 'attv3';
let data = JSON.parse(localStorage.getItem(STORE) || '{}');
 
function save() { localStorage.setItem(STORE, JSON.stringify(data)); }
 
function todayKey() {
  const d = new Date();
  return `${d.getFullYear()}-${String(d.getMonth()+1).padStart(2,'0')}-${String(d.getDate()).padStart(2,'0')}`;
}
 
function dateKey(y, m, d) {
  return `${y}-${String(m+1).padStart(2,'0')}-${String(d).padStart(2,'0')}`;
}
 
function fmtTime(iso) {
  if (!iso) return null;
  const d = new Date(iso);
  let h = d.getHours(), m = d.getMinutes(), s = d.getSeconds();
  const ampm = h >= 12 ? 'PM' : 'AM';
  h = h % 12 || 12;
  return `${String(h).padStart(2,'0')}:${String(m).padStart(2,'0')} ${ampm}`;
}
 
// CLOCK
function tick() {
  const now = new Date();
  document.getElementById('clock-time').textContent = now.toTimeString().slice(0,8);
  document.getElementById('clock-date').textContent = now.toLocaleDateString('en-IN', { weekday:'long', year:'numeric', month:'long', day:'numeric' });
}
setInterval(tick, 1000); tick();
 
// TIMELOG
function renderTimelog() {
  const key = todayKey();
  const rec = data[key] || {};
 
  const dIn = document.getElementById('disp-in');
  const dOut = document.getElementById('disp-out');
  const area = document.getElementById('action-area');
  const chips = document.getElementById('today-chips');
 
  // Login display
  if (rec.loginTime) {
    dIn.className = 'time-display locked';
    dIn.innerHTML = fmtTime(rec.loginTime) + '<span class="lock-icon">🔒</span>';
  } else {
    dIn.className = 'time-display';
    dIn.innerHTML = '<span style="color:var(--text-faint);font-size:13px;">—</span>';
  }
 
  // Logout display
  if (rec.logoutTime) {
    dOut.className = 'time-display locked';
    dOut.innerHTML = fmtTime(rec.logoutTime) + '<span class="lock-icon">🔒</span>';
  } else {
    dOut.className = 'time-display';
    dOut.innerHTML = '<span style="color:var(--text-faint);font-size:13px;">—</span>';
  }
 
  // Action buttons
  if (rec.loginTime && rec.logoutTime) {
    area.innerHTML = '<div class="locked-notice">Log complete.<br>Cannot be changed.</div>';
  } else if (rec.loginTime && !rec.logoutTime) {
    area.innerHTML = '<button class="btn btn-out" onclick="doLogout()">Log out now</button><div style="font-size:10px;color:var(--text-muted);margin-top:4px;">Login locked at ' + fmtTime(rec.loginTime) + '</div>';
  } else {
    area.innerHTML = '<button class="btn btn-in" onclick="doLogin()">Log in now</button>';
  }
 
  // Status chips
  chips.innerHTML = '';
  if (rec.loginTime && rec.logoutTime) {
    chips.innerHTML = '<span class="status-chip chip-present">&#10003; Complete</span>';
    const dur = Math.round((new Date(rec.logoutTime) - new Date(rec.loginTime)) / 60000);
    chips.innerHTML += `<span class="status-chip chip-none">${Math.floor(dur/60)}h ${dur%60}m logged</span>`;
  } else if (rec.loginTime) {
    chips.innerHTML = '<span class="status-chip chip-active">&#9679; Currently active</span>';
  } else {
    chips.innerHTML = '<span class="status-chip chip-none">Not started</span>';
  }
}
 
function doLogin() {
  const key = todayKey();
  if (!data[key]) data[key] = {};
  if (data[key].loginTime) return;
  data[key].loginTime = new Date().toISOString();
  if (!data[key].status) data[key].status = 'present';
  save(); renderTimelog(); renderCalendar(); renderSummary();
}
 
function doLogout() {
  const key = todayKey();
  if (!data[key] || !data[key].loginTime || data[key].logoutTime) return;
  data[key].logoutTime = new Date().toISOString();
  save(); renderTimelog(); renderCalendar(); renderSummary();
}
 
// CALENDAR
let viewYear = new Date().getFullYear();
let viewMonth = new Date().getMonth();
 
const MONTHS = ['January','February','March','April','May','June','July','August','September','October','November','December'];
 
function changeMonth(dir) { viewMonth += dir; if (viewMonth > 11) { viewMonth=0; viewYear++; } if (viewMonth < 0) { viewMonth=11; viewYear--; } renderCalendar(); }
 
function renderCalendar() {
  document.getElementById('cal-month-label').textContent = MONTHS[viewMonth] + ' ' + viewYear;
  const first = new Date(viewYear, viewMonth, 1).getDay();
  const days = new Date(viewYear, viewMonth+1, 0).getDate();
  const today = new Date();
  const todayNum = today.getDate(), todayMonth = today.getMonth(), todayYear = today.getFullYear();
  const grid = document.getElementById('cal-days');
  grid.innerHTML = '';
 
  for (let i = 0; i < first; i++) {
    const el = document.createElement('div');
    el.className = 'cal-day empty-day'; el.innerHTML = '<div class="day-num"></div>';
    grid.appendChild(el);
  }
 
  for (let d = 1; d <= days; d++) {
    const key = dateKey(viewYear, viewMonth, d);
    const rec = data[key] || {};
    const isToday = d === todayNum && viewMonth === todayMonth && viewYear === todayYear;
    const isFuture = new Date(viewYear, viewMonth, d) > new Date(todayYear, todayMonth, todayNum);
 
    const el = document.createElement('div');
    el.className = 'cal-day' + (isToday ? ' today' : '') + (isFuture ? ' empty-day' : '');
 
    let inner = `<div class="day-num">${d}</div>`;
 
    if (rec.status === 'present') inner += '<div class="day-dot dot-present"></div>';
    else if (rec.status === 'holiday') inner += '<div class="day-dot dot-holiday"></div>';
    else if (rec.status === 'leave') inner += '<div class="day-dot dot-leave"></div>';
 
    if (rec.loginTime) {
      let t = '<div class="day-time">in ' + fmtTime(rec.loginTime);
      if (rec.logoutTime) t += '<br>out ' + fmtTime(rec.logoutTime);
      t += '</div>';
      inner += t;
    }
 
    el.innerHTML = inner;
    if (!isFuture) el.onclick = () => openModal(d, key, rec);
    grid.appendChild(el);
  }
}
 
// SUMMARY
function renderSummary() {
  let total = 0, present = 0, holiday = 0, leave = 0;
  const today = todayKey();
  for (const key in data) {
    const rec = data[key];
    if (key > today) continue;
    if (rec.status === 'present') { total++; present++; }
    else if (rec.status === 'holiday') { total++; holiday++; }
    else if (rec.status === 'leave') { total++; leave++; }
  }
  document.getElementById('sum-total').textContent = total;
  document.getElementById('sum-present').textContent = present;
  document.getElementById('sum-holiday').textContent = holiday;
  document.getElementById('sum-leave').textContent = leave;
}
 
// MODAL
let modalKey = null, modalStatus = '';
 
function openModal(d, key, rec) {
  modalKey = key;
  modalStatus = rec.status || '';
  const label = new Date(viewYear, viewMonth, d).toLocaleDateString('en-IN', { weekday:'long', year:'numeric', month:'long', day:'numeric' });
  document.getElementById('modal-title').textContent = 'Mark Day';
  document.getElementById('modal-date-label').textContent = label;
 
  ['present','holiday','leave'].forEach(s => {
    document.getElementById('sbtn-' + s).classList.toggle('active', modalStatus === s);
  });
 
  const noteArea = document.getElementById('modal-note-area');
  if (rec.loginTime && rec.logoutTime) {
    noteArea.innerHTML = `<div class="modal-note">Time log is locked for this day.<br>Login: ${fmtTime(rec.loginTime)} &nbsp;|&nbsp; Logout: ${fmtTime(rec.logoutTime)}</div>`;
  } else if (rec.loginTime) {
    noteArea.innerHTML = `<div class="modal-note">Login locked at ${fmtTime(rec.loginTime)}</div>`;
  } else { noteArea.innerHTML = ''; }
 
  document.getElementById('modal-overlay').classList.add('open');
}
 
function selectStatus(s) {
  modalStatus = s;
  ['present','holiday','leave'].forEach(id => {
    const el = document.getElementById('sbtn-' + id);
    if (el) el.classList.toggle('active', modalStatus === id);
  });
}
 
function saveModal() {
  if (!modalKey) return;
  if (!data[modalKey]) data[modalKey] = {};
  data[modalKey].status = modalStatus;
  save(); renderCalendar(); renderSummary();
  closeModal();
}
 
function closeModal(e) {
  if (e && e.target !== document.getElementById('modal-overlay')) return;
  document.getElementById('modal-overlay').classList.remove('open');
  modalKey = null;
}
